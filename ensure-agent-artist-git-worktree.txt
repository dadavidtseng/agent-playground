```powershell
# Git Worktree Setup Script for agent-artist
# Task ID: 9b4cd616-944a-4eb7-bfe0-09784bccbd18
# Purpose: Ensure git worktree exists at specified path, is on main branch, and is clean

$WorktreePath = "C:\GitHub\agent-playground-artist"
$Branch = "main"
$RemoteBranch = "origin/main"

Write-Host "========================================" -ForegroundColor Cyan
Write-Host "🎨 Artist Agent - Git Worktree Setup" -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Cyan
Write-Host ""

# Step 1: Check if worktree path exists
Write-Host "📁 Checking worktree path: $WorktreePath" -ForegroundColor Yellow

if (Test-Path $WorktreePath) {
    Write-Host "✓ Path exists" -ForegroundColor Green
    
    # Verify it's a git worktree
    Push-Location $WorktreePath
    $isGitRepo = git rev-parse --is-inside-work-tree 2>$null
    
    if ($isGitRepo -eq "true") {
        Write-Host "✓ Valid git worktree detected" -ForegroundColor Green
    } else {
        Write-Host "✗ Path exists but is not a git worktree" -ForegroundColor Red
        Pop-Location
        exit 1
    }
} else {
    Write-Host "⚠ Worktree does not exist. Creating..." -ForegroundColor Yellow
    
    # Find main repository to create worktree from
    $MainRepoPath = "C:\GitHub\agent-playground"
    
    if (Test-Path $MainRepoPath) {
        Push-Location $MainRepoPath
        
        # Fetch latest from remote
        Write-Host "📥 Fetching from remote..." -ForegroundColor Yellow
        git fetch origin
        
        # Create worktree
        Write-Host "🔨 Creating worktree at $WorktreePath" -ForegroundColor Yellow
        git worktree add $WorktreePath $Branch
        
        if ($LASTEXITCODE -eq 0) {
            Write-Host "✓ Worktree created successfully" -ForegroundColor Green
            Pop-Location
            Push-Location $WorktreePath
        } else {
            Write-Host "✗ Failed to create worktree" -ForegroundColor Red
            Pop-Location
            exit 1
        }
    } else {
        Write-Host "✗ Main repository not found at $MainRepoPath" -ForegroundColor Red
        exit 1
    }
}

# Step 2: Check current branch
Write-Host ""
Write-Host "🌿 Checking current branch..." -ForegroundColor Yellow
$CurrentBranch = git branch --show-current

if ($CurrentBranch -eq $Branch) {
    Write-Host "✓ On correct branch: $Branch" -ForegroundColor Green
} else {
    Write-Host "⚠ Currently on branch: $CurrentBranch. Switching to $Branch..." -ForegroundColor Yellow
    
    # Check if branch exists locally
    $BranchExists = git branch --list $Branch
    
    if ($BranchExists) {
        git checkout $Branch
    } else {
        # Check if branch exists on remote
        git fetch origin
        $RemoteBranchExists = git branch -r --list $RemoteBranch
        
        if ($RemoteBranchExists) {
            Write-Host "📥 Creating local branch from $RemoteBranch" -ForegroundColor Yellow
            git checkout -b $Branch --track $RemoteBranch
        } else {
            Write-Host "✗ Branch $Branch does not exist locally or on remote" -ForegroundColor Red
            Pop-Location
            exit 1
        }
    }
    
    if ($LASTEXITCODE -eq 0) {
        Write-Host "✓ Switched to branch: $Branch" -ForegroundColor Green
    } else {
        Write-Host "✗ Failed to switch branch" -ForegroundColor Red
        Pop-Location
        exit 1
    }
}

# Step 3: Sync with remote
Write-Host ""
Write-Host "🔄 Syncing with remote..." -ForegroundColor Yellow
git fetch origin

# Check if we're behind remote
$LocalCommit = git rev-parse HEAD
$RemoteCommit = git rev-parse $RemoteBranch 2>$null

if ($RemoteCommit) {
    if ($LocalCommit -eq $RemoteCommit) {
        Write-Host "✓ Already up-to-date with $RemoteBranch" -ForegroundColor Green
    } else {
        Write-Host "⚠ Local branch is behind remote. Pulling changes..." -ForegroundColor Yellow
        git pull origin $Branch
        
        if ($LASTEXITCODE -eq 0) {
            Write-Host "✓ Successfully synced with remote" -ForegroundColor Green
        } else {
            Write-Host "✗ Failed to pull from remote" -ForegroundColor Red
            Pop-Location
            exit 1
        }
    }
} else {
    Write-Host "⚠ Remote branch $RemoteBranch not found" -ForegroundColor Yellow
}

# Step 4: Check for uncommitted changes
Write-Host ""
Write-Host "🔍 Checking working tree status..." -ForegroundColor Yellow
$Status = git status --porcelain

if ([string]::IsNullOrWhiteSpace($Status)) {
    Write-Host "✓ Working tree is clean" -ForegroundColor Green
} else {
    Write-Host "⚠ Uncommitted changes detected:" -ForegroundColor Yellow
    git status --short
    Write-Host ""
    Write-Host "✗ Working tree is not clean. Please commit or stash changes." -ForegroundColor Red
    Pop-Location
    exit 1
}

# Step 5: Verification Summary
Write-Host ""
Write-Host "========================================" -ForegroundColor Cyan
Write-Host "✅ VERIFICATION SUMMARY" -ForegroundColor Green
Write-Host "========================================" -ForegroundColor Cyan
Write-Host "📍 Worktree Path: $WorktreePath" -ForegroundColor White
Write-Host "🌿 Current Branch: $(git branch --show-current)" -ForegroundColor White
Write-Host "🔗 Tracking: $(git rev-parse --abbrev-ref --symbolic-full-name @{u} 2>$null)" -ForegroundColor White
Write-Host "📝 Working Tree: Clean" -ForegroundColor White
Write-Host "🎯 Status: Ready for agent-artist operations" -ForegroundColor Green
Write-Host "========================================" -ForegroundColor Cyan

Pop-Location

Write-Host ""
Write-Host "🎨 Artist Agent worktree setup complete! 🎨" -ForegroundColor Cyan
exit 0
```
