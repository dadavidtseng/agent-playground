import os
import hashlib

def verify_file():
    file_path = r"C:\GitHub\agent-playground\test.txt"
    
    # Read file in binary mode
    with open(file_path, 'rb') as f:
        content_bytes = f.read()
    
    # Get file size
    file_size = len(content_bytes)
    
    # Calculate checksums
    md5_hash = hashlib.md5(content_bytes).hexdigest()
    sha1_hash = hashlib.sha1(content_bytes).hexdigest()
    
    # Expected content with CRLF
    expected_content = b'This is a test file.\r\n'
    
    # Verify exact content match
    content_matches = content_bytes == expected_content
    
    # Verify CRLF line endings (0x0D 0x0A)
    has_crlf = b'\r\n' in content_bytes
    
    # Check for the exact line
    lines = content_bytes.split(b'\r\n')
    has_exact_line = lines[0] == b'This is a test file.'
    has_no_extra_lines = len([line for line in lines if line]) == 1
    
    # Verification report
    print("=" * 60)
    print("FILE VERIFICATION REPORT")
    print("=" * 60)
    print(f"File Path: {file_path}")
    print(f"File Size: {file_size} bytes")
    print(f"MD5 Checksum: {md5_hash}")
    print(f"SHA-1 Checksum: {sha1_hash}")
    print("-" * 60)
    print(f"Content Matches Expected: {content_matches}")
    print(f"Has CRLF Line Endings (0x0D 0x0A): {has_crlf}")
    print(f"Contains Exact Line 'This is a test file.': {has_exact_line}")
    print(f"No Extra Lines or Trailing Content: {has_no_extra_lines}")
    print("-" * 60)
    print(f"Raw Bytes (hex): {content_bytes.hex()}")
    print(f"Expected Bytes (hex): {expected_content.hex()}")
    print("=" * 60)
    
    # Overall verification
    if content_matches and has_crlf and has_exact_line and has_no_extra_lines:
        print("✓ VERIFICATION PASSED: File meets all requirements")
    else:
        print("✗ VERIFICATION FAILED: File does not meet requirements")
    
    return content_matches

if __name__ == "__main__":
    verify_file()
