# Directory Creation Script for AGENTS Project
# Task ID: 06a6b690-a50c-486f-b9c4-f6d3f5b132bf
# Purpose: Ensure C:\GitHub\AGENTS directory exists with proper permissions

$targetPath = "C:\GitHub\AGENTS"

try {
    # Check if directory exists
    if (-not (Test-Path -Path $targetPath -PathType Container)) {
        Write-Host "Directory does not exist. Creating: $targetPath" -ForegroundColor Yellow
        
        # Create directory with proper error handling
        New-Item -Path $targetPath -ItemType Directory -Force | Out-Null
        
        Write-Host "Successfully created directory: $targetPath" -ForegroundColor Green
        
        # Verify write permissions for current user
        $acl = Get-Acl -Path $targetPath
        $currentUser = [System.Security.Principal.WindowsIdentity]::GetCurrent().Name
        
        Write-Host "Directory created with default permissions for user: $currentUser" -ForegroundColor Cyan
    }
    else {
        Write-Host "Directory already exists: $targetPath" -ForegroundColor Green
    }
    
    # Final verification
    if (Test-Path -Path $targetPath -PathType Container) {
        $testFile = Join-Path -Path $targetPath -ChildPath ".write_test_temp"
        try {
            # Test write permission
            New-Item -Path $testFile -ItemType File -Force | Out-Null
            Remove-Item -Path $testFile -Force
            Write-Host "Write permission verified successfully." -ForegroundColor Green
            Write-Host "SUCCESS: Task completed successfully." -ForegroundColor Green
        }
        catch {
            Write-Host "WARNING: Directory exists but write permission verification failed." -ForegroundColor Red
            Write-Host "Error: $_" -ForegroundColor Red
        }
    }
}
catch {
    Write-Host "ERROR: Failed to create directory." -ForegroundColor Red
    Write-Host "Error Details: $_" -ForegroundColor Red
    exit 1
}
