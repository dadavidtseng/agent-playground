# Directory Verification Script for C:\GitHub\agent-playground
# Artist Agent - Creative Systems Verification

$targetPath = "C:\GitHub\agent-playground"
$timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"

Write-Host "========================================" -ForegroundColor Cyan
Write-Host "  DIRECTORY VERIFICATION PROTOCOL" -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Cyan
Write-Host ""

# Test if path exists
Write-Host "[CHECK 1] Testing Path Existence..." -ForegroundColor Yellow
if (Test-Path -Path $targetPath) {
    Write-Host "✓ SUCCESS: Directory exists at $targetPath" -ForegroundColor Green
    
    # Get directory details
    $dirInfo = Get-Item -Path $targetPath
    Write-Host ""
    Write-Host "[CHECK 2] Directory Properties:" -ForegroundColor Yellow
    Write-Host "  - Full Path: $($dirInfo.FullName)" -ForegroundColor White
    Write-Host "  - Created: $($dirInfo.CreationTime)" -ForegroundColor White
    Write-Host "  - Last Modified: $($dirInfo.LastWriteTime)" -ForegroundColor White
    Write-Host "  - Attributes: $($dirInfo.Attributes)" -ForegroundColor White
    
    # Test write permissions
    Write-Host ""
    Write-Host "[CHECK 3] Testing Write Permissions..." -ForegroundColor Yellow
    $testFile = Join-Path $targetPath ".agent_write_test_$(Get-Random).tmp"
    try {
        New-Item -Path $testFile -ItemType File -Force | Out-Null
        Remove-Item -Path $testFile -Force
        Write-Host "✓ SUCCESS: Write permissions verified" -ForegroundColor Green
    } catch {
        Write-Host "✗ ERROR: Write permission denied" -ForegroundColor Red
        Write-Host "  Details: $($_.Exception.Message)" -ForegroundColor Red
        exit 1
    }
    
    # Test read permissions
    Write-Host ""
    Write-Host "[CHECK 4] Testing Read Permissions..." -ForegroundColor Yellow
    try {
        $contents = Get-ChildItem -Path $targetPath -ErrorAction Stop
        Write-Host "✓ SUCCESS: Read permissions verified" -ForegroundColor Green
        Write-Host "  - Items in directory: $($contents.Count)" -ForegroundColor White
    } catch {
        Write-Host "✗ ERROR: Read permission denied" -ForegroundColor Red
        Write-Host "  Details: $($_.Exception.Message)" -ForegroundColor Red
        exit 1
    }
    
    Write-Host ""
    Write-Host "========================================" -ForegroundColor Cyan
    Write-Host "  ✓ ALL CHECKS PASSED" -ForegroundColor Green
    Write-Host "  Directory is accessible and operational" -ForegroundColor Green
    Write-Host "========================================" -ForegroundColor Cyan
    Write-Host ""
    Write-Host "Timestamp: $timestamp" -ForegroundColor Gray
    exit 0
    
} else {
    Write-Host "✗ CRITICAL ERROR: Directory does not exist" -ForegroundColor Red
    Write-Host ""
    Write-Host "Expected Path: $targetPath" -ForegroundColor Yellow
    Write-Host "Status: NOT FOUND" -ForegroundColor Red
    Write-Host ""
    Write-Host "========================================" -ForegroundColor Red
    Write-Host "  VERIFICATION FAILED" -ForegroundColor Red
    Write-Host "  Stopping automated file creation" -ForegroundColor Red
    Write-Host "========================================" -ForegroundColor Red
    Write-Host ""
    Write-Host "Action Required:" -ForegroundColor Yellow
    Write-Host "1. Create directory: New-Item -Path '$targetPath' -ItemType Directory" -ForegroundColor White
    Write-Host "2. Verify path spelling and drive letter" -ForegroundColor White
    Write-Host "3. Check network connectivity (if network path)" -ForegroundColor White
    Write-Host ""
    Write-Host "Timestamp: $timestamp" -ForegroundColor Gray
    exit 1
}
